
\section*{Álgebra Booleana}

\begin{minipage}{0.5\textwidth}
\begin{center} 
    Teoremas de Múltiples Variables
    \begin{align*}
        x + y &= y + x \\
        x \cdot y &= y \cdot x \\
        x + (y + z) &= x + y+ z \\
        x(yz) &= xyz \\
        x(y+z) &= xy + xz \\
        (w+x) (y+z) &= wy + wz + xy + xz \\
        x + xy &= x \\
        x + \bar x y &= x + y \\
        \bar x + xy &= \bar x + y
    \end{align*}
\end{center}
\end{minipage}
\begin{minipage}{0.5\textwidth}
\begin{center}
   Teoremas de Morgan
    \begin{align*}
        \overline{ x + y } &= \bar x \cdot \bar y \\
        \overline{ x \cdot y } &= \bar x + \bar y
    \end{align*} 
\end{center}
\begin{center}
    Teoremas de una Variable
    \begin{align*}
                 O&R       &        AN&D        \\
        x \cdot 0 &= 0     &    x + 0 &= x      \\
        x \cdot 1 &= x     &    x + 1 &= 1      \\
        x \cdot x &= x     &    x + x &= x      \\
        x \cdot\bar x &= 0 &    x + \bar x &= 1 
    \end{align*}
\end{center}
\end{minipage}
